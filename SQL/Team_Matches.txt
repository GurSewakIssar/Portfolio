-- Method 1
with cte as
(
	select row_number() over (order by team_name) rnm ,t.* from #teams t
)

select c.* from cte c
join cte c2
on c.rnm< c2.rnm


-- Method 2
select t.team_name, tt.team_name from #teams t, #teams tt
where t.team_name> tt.team_name
