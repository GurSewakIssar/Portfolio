-- Return all the 3 or more consecutive Free/Empty seats
-- Method 1
select seat_id from
(
	select *, count(grp) over (partition by grp) cnt from
	(
		select *, 
		row_number() over (order by seat_id) - 
		row_number() over (partition by free order by seat_id) grp
		from #cinema
	) as foo
	where free = 1
) as foo2
where cnt>2

-- Method 2

select seat_id from
(
	select *,
	case when free = 1 and lag(free,1) over (order by seat_id) = 1 and lag(free,2) over (order by seat_id) = 1 then 1 else
	case when free = 1 and lead(free,1) over (order by seat_id) = 1 and lead(free,2) over (order by seat_id) = 1 then 1 else
	case when free = 1 and lag(free,1) over (order by seat_id) = 1 and lead(free,1) over (order by seat_id) = 1 then 1 else
	0 end end end flag
	from #cinema
) as foo
where flag = 1

-- Method 3
select seat_id from
(
	select *, count(grp) over (partition by grp) cnt
	from
	(
		select *,
		seat_id - row_number() over (partition by free order by seat_id) grp
		from #cinema
	) as foo
	where free = 1
) as foo2
where cnt>2
