/*
Return User_Id and Page_id for the user whose friend has liked a page which the user has not liked yet.
*/

CREATE TABLE #friends (
    user_id INT,
    friend_id INT
);

-- Create #likes table
CREATE TABLE #likes (
    user_id INT,
    page_id CHAR(1)
);

select * from #friends
select * from #likes;

-- Method 1
with cte as
(
	select f.*, l.page_id from #friends f
	join #likes l 
	on f.user_id = l.user_id
)
select distinct coalesce(c.user_id, cc.friend_id) id, cc.page_id from cte c
full join cte cc
on c.user_id = cc.friend_id and
c.page_id = cc.page_id
where c.user_id is null


-- Method 2

select distinct f.user_id, l.page_id  from #friends f
join #likes l
on f.friend_id = l.user_id 
left join #likes ll
on ll.user_id = f.user_id and
l.page_id = ll.page_id
where ll.user_id is null
