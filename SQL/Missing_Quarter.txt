-- FIND THE QUARTER WHEN STORE IS OFF FOR EACH STORE


CREATE TABLE #STORES (
Store varchar(10),
Quarter varchar(10),
Amount int);

INSERT INTO #STORES (Store, Quarter, Amount)
VALUES ('S1', 'Q1', 200),
('S1', 'Q2', 300),
('S1', 'Q4', 400),
('S2', 'Q1', 500),
('S2', 'Q3', 600),

-- METHOD 1
SELECT STORE,CASE WHEN 'Q1' NOT IN (SELECT DISTINCT(QUARTER) FROM #STORES S WHERE M.STORE = S.STORE) THEN 'Q1'
WHEN 'Q2' NOT IN (SELECT DISTINCT(QUARTER) FROM #STORES S WHERE M.STORE = S.STORE) THEN 'Q2'
WHEN 'Q3' NOT IN (SELECT DISTINCT(QUARTER) FROM #STORES S WHERE M.STORE = S.STORE) THEN 'Q3' 
WHEN 'Q4' NOT IN (SELECT DISTINCT(QUARTER) FROM #STORES S WHERE M.STORE = S.STORE) THEN 'Q4' END 
FROM #STORES M
GROUP BY STORE;



-- METHOD 2

-- HERE IF WE SUBTRACT FROM 10, THE SUM OF QUARTER VALUES WE GONNA GET MISSING QTR,
-- ONLY WORKS WHEN AMONG 4 ONLY 1 IS MISSING AS PROBLEM STATES

SELECT STORE , 'Q' + CAST((10-(SUM( CAST ((RIGHT(QUARTER,1)) AS INT)))) AS VARCHAR) MISSING_QTR FROM #STORES
GROUP BY STORE

WITH RECURSIVE AS (
	SELECT DISTINCT Quarter QTR, 1 N, 'S'+CAST( 1 AS VARCHAR)  STORE FROM #STORES
	UNION ALL
	SELECT QTR, N+1, 'S'+ CAST(N+1 AS VARCHAR) FROM RECURSIVE
	WHERE N+1<4)

SELECT R.STORE, R.QTR FROM RECURSIVE R
LEFT JOIN #STORES S
ON R.STORE = S.STORE AND R.QTR = S.Quarter
WHERE S.Quarter IS NULL;
