-- Method 1
select sms_date,#users, max(sender) frst, min(receiver) second , sum(sms_no) #sms from 
	(
	select *, case when sender > receiver then sender+'-'+receiver else receiver+'-'+sender end #users from #subscriber
	) a
group by sms_date, #users



-- Method 2
select case when sender < receiver then sender else receiver end sender,
case when sender > receiver then sender else receiver end receiver, sum(sms_no) [Msgs_Exchanged]
from #subscriber
group by case when sender < receiver then sender else receiver end ,
