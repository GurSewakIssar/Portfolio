/*
PROBLEM STATEMENT: Given tables represent the marks scored by engineering students.
Create a report to display the following results for each student.
  - Student_id, Student name
  - Total Percentage of all marks
  - Failed subjects (must be comma seperated values in case of multiple failed subjects)
  - Result (if percentage >= 70% then 'First Class', if >= 50% & <=70% then 'Second class', if <=50% then 'Third class' else 'Fail'.
  			The result should be Fail if a students fails in any subject irrespective of the percentage marks)
	
	*** The sequence of subjects in #sutdent_marks table match with the sequential id from #subjects table.
	*** students have the option to choose either 4 or 5 subjects only.
*/

-- Solution

with cte as
(
	select student_id, col as subjects, marks
	from
	(
		select * from #sutdent_marks
	) as pt
	unpivot
	(
		marks for col in 
		([subject1], [subject2], [subject3], [subject4], [subject5], [subject6])
	)as pt2
),
student_stats as
(
	select *, sum(marks) over (partition by student_id)*1. / count(student_id) 
	over (partition by student_id) prcnt_marks,
	case when marks < pass_marks then [name] end flag
	from cte c
	join #subjects s
	on concat(left(c.subjects,1),right(c.subjects,1)) = s.id
)
select student_id, st.name,  max(prcnt_marks) marks_percentage,
coalesce(string_agg(flag, ',') within group (order by flag), '-') Failed_subjects,
case when count(flag) > 0 then 'Fail' else 
case when max(prcnt_marks)>= 70 then 'First' else
case when max(prcnt_marks)>= 50 then 'Second' else
'Third' end end end Divison
from student_stats s
join #students st
on s.student_id = st.roll_no
group by student_id, st.name
order by student_id
