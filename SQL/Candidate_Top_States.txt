select concat(max(first_name), ' ', max(last_name)) [Name],
string_agg(case when rnk=1 then concat(state, ' (', cast(cnt as varchar(max)), ')') end, ', ') within group (order by state) First_,
string_agg(case when rnk=2 then concat(state, ' (', cast(cnt as varchar(max)), ')') end, ', ') within group (order by state) Second_,
string_agg(case when rnk=3 then concat(state, ' (', cast(cnt as varchar(max)), ')') end, ', ') within group (order by state) Third_
from
(
select r.candidate_id, c.first_name, c.last_name, r.state,
dense_rank() over (partition by candidate_id order by count(state) desc) rnk, count(state) cnt
from #results_tab r
join #candidates_tab c
on c.id = r.candidate_id
group by r.candidate_id, c.first_name, c.last_name, r.state
) as foo
group by candidate_id
